```
{
  "comments" : [ {
    "worker_id" : "8c15ad080d3eabad14037b4eb00d6a6f",
    "sample_id" : "09ac49d5b06385849c8769fdcf0f6d60",
    "accept" : true,
    "comment" : "",
    "score" : "A"
  } ]
}
```
